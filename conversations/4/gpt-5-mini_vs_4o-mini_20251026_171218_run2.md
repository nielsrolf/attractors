(async () => {
  const browser = await puppeteer.launch({ args: ['--no-sandbox', '--disable-setuid-sandbox'] });
  try {
    for (const customer of customers) {
      const data = {
        CUSTOMER_NAME: customer.customer_name || customer.customer_company || 'customer',
        CUSTOMER_COMPANY: customer.customer_company || '',
        DATE: customer.date || new Date().toISOString().split('T')[0],
        CALENDLY_LINK: customer.calendly_link || ''
      };

      for (const tpl of (customer.templates || [])) {
        const templateFile = templates[tpl];
        if (!templateFile || !fs.existsSync(templateFile)) {
          console.warn(`Template ${tpl} not found or template file missing: ${templateFile}`);
          continue;
        }

        // Read template HTML
        let html = fs.readFileSync(path.join(__dirname, templateFile), 'utf8');

        // Replace tokens like {{CUSTOMER_NAME}} etc.
        html = html.replace(/{{\s*CUSTOMER_NAME\s*}}/g, data.CUSTOMER_NAME);
        html = html.replace(/{{\s*CUSTOMER_COMPANY\s*}}/g, data.CUSTOMER_COMPANY);
        html = html.replace(/{{\s*DATE\s*}}/g, data.DATE);
        html = html.replace(/{{\s*CALENDLY_LINK\s*}}/g, data.CALENDLY_LINK);

        // Write a temp HTML file
        const tmpHtmlPath = path.join(__dirname, `.tmp_${tpl}_${sanitize(data.CUSTOMER_NAME)}.html`);
        fs.writeFileSync(tmpHtmlPath, html, 'utf8');

        // Render PDF
        const page = await browser.newPage();
        await page.goto('file://' + tmpHtmlPath, { waitUntil: 'networkidle0' });

        const friendlyName = sanitize(data.CUSTOMER_NAME || 'customer');
        const outName = `${tpl === 'sow' ? 'Startup-Pilot-SOW' : 'Startup-Pilot'}-${friendlyName}.pdf`;
        const outPath = path.join(__dirname, outName);

        await page.pdf({
          path: outPath,
          format: 'Letter', // or 'A4'
          printBackground: true,
          margin: { top: '12mm', bottom: '12mm', left: '10mm', right: '10mm' }
        });

        console.log('Generated:', outPath);

        await page.close();
        fs.unlinkSync(tmpHtmlPath); // cleanup
      }
    }
  } catch (err) {
    console.error('Error:', err);
  } finally {
    await browser.close();
  }
})();

const fs = require('fs');
const path = require('path');
const puppeteer = require('puppeteer');

const templates = {
  onepager: 'pilot-onepager.html',
  sow: 'pilot-sow.html'
};

function sanitize(s) {
  if (!s) return 'customer';
  return s.replace(/[^a-z0-9_\-\.]/gi, '_').replace(/_+/g, '_').toLowerCase();
}

function loadJSON(p) {
  if (!fs.existsSync(p)) { console.error('Missing', p); process.exit(1); }
  return JSON.parse(fs.readFileSync(p, 'utf8'));
}

const customers = loadJSON(path.join(__dirname, 'customers.json'));

// Optional global defaults
const GLOBAL = {
  YOUR_NAME: process.env.YOUR_NAME || 'Your Name',
  YOUR_EMAIL: process.env.YOUR_EMAIL || 'you@startup.com',
};

(async () => {
  const browser = await puppeteer.launch({ args: ['--no-sandbox', '--disable-setuid-sandbox'] });
  try {
    for (const customer of customers) {
      const data = {
        CUSTOMER_NAME: customer.customer_name || '',
        CUSTOMER_COMPANY: customer.customer_company || '',
        DATE: customer.date || '',
        CALENDLY_LINK: customer.calendly_link || '',
        YOUR_NAME: customer.your_name || GLOBAL.YOUR_NAME,
        YOUR_EMAIL: customer.your_email || GLOBAL.YOUR_EMAIL,
        PILOT_DURATION: customer.pilot_duration || '',
        PILOT_SCOPE: customer.pilot_scope || '',
        PILOT_SUCCESS_METRIC: customer.pilot_success_metric || ''
      };

      for (const tpl of (customer.templates || [])) {
        const tplFile = templates[tpl];
        const tplPath = path.join(__dirname, tplFile);
        if (!tplFile || !fs.existsSync(tplPath)) {
          console.warn(`Template ${tplFile} not found; skipping for ${data.CUSTOMER_NAME}`);
          continue;
        }

        let html = fs.readFileSync(tplPath, 'utf8');

        // Replace tokens
        Object.keys(data).forEach(key => {
          const re = new RegExp(`{{\\s*${key}\\s*}}`, 'g');
          html = html.replace(re, data[key]);
        });

        // Temp HTML path
        const friendly = sanitize(data.CUSTOMER_NAME || data.CUSTOMER_COMPANY || 'customer');
        const tmpHtml = path.join(__dirname, `.tmp_${tpl}_${friendly}.html`);
        fs.writeFileSync(tmpHtml, html, 'utf8');

        const page = await browser.newPage();
        await page.goto('file://' + tmpHtml, { waitUntil: 'networkidle0' });

        const outName = `${tpl === 'sow' ? 'Startup-Pilot-SOW' : 'Startup-Pilot'}-${friendly}.pdf`;
        const outPath = path.join(__dirname, outName);

        await page.pdf({
          path: outPath,
          format: 'Letter', // change to 'A4' if you prefer
          printBackground: true,

const puppeteer = require('puppeteer');

const templates = {
  onepager: 'pilot-onepager.html',
  sow: 'pilot-sow.html'

function sanitize(s) {
  if (!s) return 'customer';
  return s.replace(/[^a-z0-9_\-\.]/gi, '_').replace(/_+/g, '_').toLowerCase();

function loadJSON(p) {

}

const customers = loadJSON(path.join(__dirname, 'customers.json'));

// Optional global defaults
const GLOBAL = {
  YOUR_NAME: process.env.YOUR_NAME || 'Your Name',
  YOUR_EMAIL: process.env.YOUR_EMAIL || 'you@startup.com',
};

const CONCURRENCY = parseInt(process.env.CONCURRENCY || '3', 10);

(async () => {
  const browser = await puppeteer.launch({ args: ['--no-sandbox', '--disable-setuid-sandbox'] });

  try {
    // Build tasks: list of { tpl, data }
    const tasks = [];
    for (const customer of customers) {
      const data = {
        CUSTOMER_NAME: customer.customer_name || '',
        CUSTOMER_COMPANY: customer.customer_company || '',
        DATE: customer.date || '',
        CALENDLY_LINK: customer.calendly_link || '',

        PILOT_SCOPE: customer.pilot_scope || '',
        PILOT_SUCCESS_METRIC: customer.pilot_success_metric || ''
      };
      for (const tpl of (customer.templates || [])) {
        tasks.push({ tpl, data });
      }
    }

    console.log(`Built ${tasks.length} render tasks. Running with concurrency=${CONCURRENCY}`);

    // Process in batches of size CONCURRENCY
    for (let i = 0; i < tasks.length; i += CONCURRENCY) {
      const batch = tasks.slice(i, i + CONCURRENCY);
      await Promise.all(batch.map(async task => {
        const tplFile = templates[task.tpl];
        const tplPath = path.join(__dirname, tplFile);
        if (!tplFile || !fs.existsSync(tplPath)) {
          console.warn(`Template ${tplFile} not found; skipping`);
          return;
        }

        let html = fs.readFileSync(tplPath, 'utf8');

        // Replace tokens
        Object.keys(task.data).forEach(key => {
          const re = new RegExp(`{{\\s*${key}\\s*}}`, 'g');
          html = html.replace(re, task.data[key]);
        });

        // Temp HTML path
        const friendly = sanitize(task.data.CUSTOMER_NAME || task.data.CUSTOMER_COMPANY || 'customer');
        const tmpHtml = path.join(__dirname, `.tmp_${task.tpl}_${friendly}.html`);
        fs.writeFileSync(tmpHtml, html, 'utf8');

        const page = await browser.newPage();
        try {

async function loadCustomers() {
  const csvPath = path.join(__dirname, 'customers.csv');
  const jsonPath = path.join(__dirname, 'customers.json');

  if (fs.existsSync(csvPath)) {

    const rows = await csv().fromFile(csvPath);
    // Normalize rows: ensure fields exist and parse templates into array
    return rows.map(r => ({

      customer_company: r.customer_company || r.company || '',
      date: r.date || '',
      calendly_link: r.calendly_link || r.calendly || '',
